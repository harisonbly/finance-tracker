document.addEventListener("DOMContentLoaded", async () => { const hashParams = new URLSearchParams(window.location.hash.substring(1)); const accessToken = hashParams.get("access_token"); const refreshToken = hashParams.get("refresh_token"); if (accessToken) { await supabase.auth.setSession({ access_token: accessToken, refresh_token: refreshToken }); window.location.href = "dashboard.html"; } });
